182 def is_distro_package(self, package): 

183 '''Check if a package is a genuine distro package. 

184 

185 Return True for a native distro package, False if it comes from a 

186 third-party source. 

187 ''' 

188 pkg = self._apt_pkg(package) 

189 # some PPA packages have installed version None, see LP#252734 

190 if pkg.installed and pkg.installed.version is None: 

191 return False 

192 

193 distro_name = self.get_os_version()[0] 

194 

195 if pkg.candidate and pkg.candidate.origins: # might be None 

196 for o in pkg.candidate.origins: 

197 if o.origin == distro_name: 

198 return True 

199 

200 # on Ubuntu system-image we might not have any /var/lib/apt/lists 

201 if set([o.origin for o in pkg.candidate.origins]) == set(['']) and \ 

202 os.path.exists('/etc/system-image/channel.ini'): 

203 return True 

204 

205 return False

270 def json_request(self, url, entries=False): 

271 '''Open, read and parse the json of a url 

272 

273 Set entries to True when the json data returned by Launchpad 

274 has a dictionary with an entries key which contains the data 

275 desired. 

276 ''' 

277 try: 

278 response = urlopen(url) 

279 if response.getcode() >= 400: 

280 raise HTTPError('%u' % response.getcode()) 

281 except (URLError, HTTPError): 

282 apport.warning('cannot connect to: %s' % unquote(url)) 

283 return None 

284 try: 

285 content = response.read() 

286 except IOError: 

287 apport.warning('failure reading data at: %s' % unquote(url)) 

288 return None 

289 if isinstance(content, bytes): 

290 content = content.decode('utf-8') 

291 if entries: 

292 return json.loads(content)['entries'] 

293 else: 

294 return json.loads(content)

296 def get_lp_source_package(self, distro_id, package, version): 

297 package = quote(package) 

298 version = quote(version) 

299 ma = self.json_request(self._archive_url % distro_id) 

300 if not ma: 

301 return None 

302 ma_link = ma['self_link'] 

303 ps_url = ma_link + ('/?ws.op=getPublishedSources&exact_match=true&source_name=%s&version=%s' % 

304 (package, version)) 

305 # use the first entry as they are sorted chronologically 

306 try: 

307 ps = self.json_request(ps_url, entries=True)[0]['self_link'] 

308 except IndexError: 

309 return None 

310 if not ps: 

311 return None 

312 sf_urls = ps + '?ws.op=sourceFileUrls' 

313 sfus = self.json_request(sf_urls) 

314 if not sfus: 

315 return None 

316 

317 source_files = [] 

318 for sfu in sfus: 

319 if sys.version_info.major == 2 and isinstance(sfu, unicode): 

320 sfu = sfu.encode('utf-8') 

321 sfu = unquote(sfu) 

322 source_files.append(sfu) 

323 

324 return source_files

347 def get_modified_files(self, package): 

348 '''Return list of all modified files of a package.''' 

349 

350 # get the maximum mtime of package files that we consider unmodified 

351 listfile = '/var/lib/dpkg/info/%s:%s.list' % (package, self.get_system_architecture()) 

352 if not os.path.exists(listfile): 

353 listfile = '/var/lib/dpkg/info/%s.list' % package 

354 try: 

355 s = os.stat(listfile) 

356 if not stat.S_ISREG(s.st_mode): 

357 raise OSError 

358 max_time = max(s.st_mtime, s.st_ctime) 

359 except OSError: 

360 return [] 

361 

362 # create a list of files with a newer timestamp for md5sum'ing 

363 sums = b'' 

364 sumfile = '/var/lib/dpkg/info/%s:%s.md5sums' % (package, self.get_system_architecture()) 

365 if not os.path.exists(sumfile): 

366 sumfile = '/var/lib/dpkg/info/%s.md5sums' % package 

367 if not os.path.exists(sumfile): 

368 # some packages do not ship md5sums 

369 return [] 

370 

371 with open(sumfile, 'rb') as fd: 

372 for line in fd: 

373 try: 

374 # ignore lines with NUL bytes (happens, LP#96050) 

375 if b'\0' in line: 

376 apport.warning('%s contains NUL character, ignoring line', sumfile) 

377 continue 

378 words = line.split() 

379 if not words: 

380 apport.warning('%s contains empty line, ignoring line', sumfile) 

381 continue 

382 s = os.stat(('/' + words[-1].decode('UTF-8')).encode('UTF-8')) 

383 if max(s.st_mtime, s.st_ctime) <= max_time: 

384 continue 

385 except OSError: 

386 pass 

387 

388 sums += line 

389 

390 if sums: 

391 return self._check_files_md5(sums) 

392 else: 

393 return []

395 def get_modified_conffiles(self, package): 

396 '''Return modified configuration files of a package. 

397 

398 Return a file name -> file contents map of all configuration files of 

399 package. Please note that apport.hookutils.attach_conffiles() is the 

400 official user-facing API for this, which will ask for confirmation and 

401 allows filtering. 

402 ''' 

403 dpkg = subprocess.Popen(['dpkg-query', '-W', '--showformat=${Conffiles}', 

404 package], stdout=subprocess.PIPE) 

405 

406 out = dpkg.communicate()[0].decode() 

407 if dpkg.returncode != 0: 

408 return {} 

409 

410 modified = {} 

411 for line in out.splitlines(): 

412 if not line: 

413 continue 

414 # just take the first two fields, to not stumble over obsolete 

415 # conffiles 

416 path, default_md5sum = line.strip().split()[:2] 

417 

418 if os.path.exists(path): 

419 try: 

420 with open(path, 'rb') as fd: 

421 contents = fd.read() 

422 m = hashlib.md5() 

423 m.update(contents) 

424 calculated_md5sum = m.hexdigest() 

425 

426 if calculated_md5sum != default_md5sum: 

427 modified[path] = contents 

428 except IOError as e: 

429 modified[path] = '[inaccessible: %s]' % str(e) 

430 else: 

431 modified[path] = '[deleted]' 

432 

433 return modified

435 def __fgrep_files(self, pattern, file_list): 

436 '''Call fgrep for a pattern on given file list and return the first 

437 matching file, or None if no file matches.''' 

438 

439 match = None 

440 slice_size = 100 

441 i = 0 

442 

443 while not match and i < len(file_list): 

444 p = subprocess.Popen(['fgrep', '-lxm', '1', '--', pattern] + 

445 file_list[i:(i + slice_size)], stdin=subprocess.PIPE, 

446 stdout=subprocess.PIPE, stderr=subprocess.PIPE) 

447 out = p.communicate()[0].decode('UTF-8') 

448 if p.returncode == 0: 

449 match = out 

450 i += slice_size 

451 

452 return match

454 def get_file_package(self, file, uninstalled=False, map_cachedir=None, 

455 release=None, arch=None): 

456 '''Return the package a file belongs to. 

457 

458 Return None if the file is not shipped by any package. 

459 

460 If uninstalled is True, this will also find files of uninstalled 

461 packages; this is very expensive, though, and needs network access and 

462 lots of CPU and I/O resources. In this case, map_cachedir can be set to 

463 an existing directory which will be used to permanently store the 

464 downloaded maps. If it is not set, a temporary directory will be used. 

465 Also, release and arch can be set to a foreign release/architecture 

466 instead of the one from the current system. 

467 ''' 

468 if uninstalled: 

469 return self._search_contents(file, map_cachedir, release, arch) 

470 

471 # check if the file is a diversion 

472 dpkg = subprocess.Popen(['dpkg-divert', '--list', file], 

473 stdout=subprocess.PIPE, stderr=subprocess.PIPE) 

474 out = dpkg.communicate()[0].decode('UTF-8') 

475 if dpkg.returncode == 0 and out: 

476 pkg = out.split()[-1] 

477 if pkg != 'hardening-wrapper': 

478 return pkg 

479 

480 fname = os.path.splitext(os.path.basename(file))[0].lower() 

481 

482 all_lists = [] 

483 likely_lists = [] 

484 for f in glob.glob('/var/lib/dpkg/info/*.list'): 

485 p = os.path.splitext(os.path.basename(f))[0].lower().split(':')[0] 

486 if p in fname or fname in p: 

487 likely_lists.append(f) 

488 else: 

489 all_lists.append(f) 

490 

491 # first check the likely packages 

492 match = self.__fgrep_files(file, likely_lists) 

493 if not match: 

494 match = self.__fgrep_files(file, all_lists) 

495 

496 if match: 

497 return os.path.splitext(os.path.basename(match))[0].split(':')[0] 

498 

499 return None

544 def get_source_tree(self, srcpackage, dir, version=None, sandbox=None, 

545 apt_update=False): 

546 '''Download source package and unpack it into dir. 

547 

548 This also has to care about applying patches etc., so that dir will 

549 eventually contain the actually compiled source. dir needs to exist and 

550 should be empty. 

551 

552 If version is given, this particular version will be retrieved. 

553 Otherwise this will fetch the latest available version. 

554 

555 If sandbox is given, it calls apt-get source in that sandbox, otherwise 

556 it uses the system apt configuration. 

557 

558 If apt_update is True, it will call apt-get update before apt-get 

559 source. This is mostly necessary for freshly created sandboxes. 

560 

561 Return the directory that contains the actual source root directory 

562 (which might be a subdirectory of dir). Return None if the source is 

563 not available. 

564 ''' 

565 # configure apt for sandbox 

566 env = os.environ.copy() 

567 if sandbox: 

568 f = tempfile.NamedTemporaryFile() 

569 f.write(('''Dir "%s"; 

570Dir::State::Status "/var/lib/dpkg/status"; 

571Debug::NoLocking "true"; 

572 ''' % sandbox).encode()) 

573 f.flush() 

574 env['APT_CONFIG'] = f.name 

575 

576 if apt_update: 

577 subprocess.call(['apt-get', '-qq', 'update'], env=env) 

578 

579 # fetch source tree 

580 argv = ['apt-get', '-qq', '--assume-yes', 'source', srcpackage] 

581 if version: 

582 argv[-1] += '=' + version 

583 try: 

584 if subprocess.call(argv, cwd=dir, env=env) != 0: 

585 if not version: 

586 return None 

587 sf_urls = self.get_lp_source_package(self.get_distro_name(), 

588 srcpackage, version) 

589 if sf_urls: 

590 proxy = '' 

591 if apt.apt_pkg.config.find('Acquire::http::Proxy') != '': 

592 proxy = apt.apt_pkg.config.find('Acquire::http::Proxy') 

593 apt.apt_pkg.config.set('Acquire::http::Proxy', '') 

594 fetchProgress = apt.progress.base.AcquireProgress() 

595 fetcher = apt.apt_pkg.Acquire(fetchProgress) 

596 af_queue = [] 

597 for sf in sf_urls: 

598 af_queue.append(apt.apt_pkg.AcquireFile(fetcher, 

599 sf, destdir=dir)) 

600 result = fetcher.run() 

601 if result != fetcher.RESULT_CONTINUE: 

602 return None 

603 if proxy: 

604 apt.apt_pkg.config.set('Acquire::http::Proxy', proxy) 

605 for dsc in glob.glob(os.path.join(dir, '*.dsc')): 

606 subprocess.call(['dpkg-source', '-sn', 

607 '-x', dsc], stdout=subprocess.PIPE, 

608 cwd=dir) 

609 else: 

610 return None 

611 except OSError: 

612 return None 

613 

614 # find top level directory 

615 root = None 

616 for d in glob.glob(os.path.join(dir, srcpackage + '-*')): 

617 if os.path.isdir(d): 

618 root = d 

619 assert root, 'could not determine source tree root directory' 

620 

621 # apply patches on a best-effort basis 

622 try: 

623 subprocess.call('(debian/rules patch || debian/rules apply-patches ' 

624 '|| debian/rules apply-dpatches || ' 

625 'debian/rules unpack || debian/rules patch-stamp || ' 

626 'debian/rules setup) >/dev/null 2>&1', shell=True, cwd=root) 

627 except OSError: 

628 pass 

629 

630 return root

656 return (installed, outdated) 

657 target_dir = apt.apt_pkg.config.find_dir('Dir::Cache::archives') + '/partial' 

658 deb = '%s_%s_%s.ddeb' % (debug_pkgname, ver, arch) 

659 # FIXME: this package is currently not in Packages.gz 

660 url = 'http://ddebs.ubuntu.com/pool/main/l/linux/%s' % deb 

661 out = open(os.path.join(target_dir, deb), 'w') 

662 # urlretrieve does not return 404 in the headers so we use urlopen 

663 u = urlopen(url) 

664 if u.getcode() > 400: 

665 return ('', 'linux') 

666 while True: 

667 block = u.read(8 * 1024) 

668 if not block: 

669 break 

670 out.write(block) 

671 out.flush() 

672 out.close() 

673 ret = subprocess.call(['dpkg', '-i', os.path.join(target_dir, deb)]) 

674 if ret == 0: 

675 installed.append(deb.split('_')[0]) 

676 return (installed, outdated)

678 def install_packages(self, rootdir, configdir, release, packages, 

679 verbose=False, cache_dir=None, 

680 permanent_rootdir=False, architecture=None, 

681 origins=None, install_dbg=True, install_deps=False): 

682 '''Install packages into a sandbox (for apport-retrace). 

683 

684 In order to work without any special permissions and without touching 

685 the running system, this should only download and unpack packages into 

686 the given root directory, not install them into the system. 

687 

688 configdir points to a directory with by-release configuration files for 

689 the packaging system; this is completely dependent on the backend 

690 implementation, the only assumption is that this looks into 

691 configdir/release/, so that you can use retracing for multiple 

692 DistroReleases. As a special case, if configdir is None, it uses the 

693 current system configuration, and "release" is ignored. 

694 

695 release is the value of the report's 'DistroRelease' field. 

696 

697 packages is a list of ('packagename', 'version') tuples. If the version 

698 is None, it should install the most current available version. 

699 

700 If cache_dir is given, then the downloaded packages will be stored 

701 there, to speed up subsequent retraces. 

702 

703 If permanent_rootdir is True, then the sandbox created from the 

704 downloaded packages will be reused, to speed up subsequent retraces. 

705 

706 If architecture is given, the sandbox will be created with packages of 

707 the given architecture (as specified in a report's "Architecture" 

708 field). If not given it defaults to the host system's architecture. 

709 

710 If origins is given, the sandbox will be created with apt data sources 

711 for foreign origins. 

712 

713 If install_deps is True, then the dependencies of packages will also 

714 be installed. 

715 

716 Return a string with outdated packages, or an empty string if all 

717 packages were installed. 

718 

719 If something is wrong with the environment (invalid configuration, 

720 package servers down, etc.), this should raise a SystemError with a 

721 meaningful error message. 

722 ''' 

723 if not architecture: 

724 architecture = self.get_system_architecture() 

725 if not configdir: 

726 apt_sources = '/etc/apt/sources.list' 

727 self.current_release_codename = self.get_distro_codename() 

728 else: 

729 # support architecture specific config, fall back to global config 

730 apt_sources = os.path.join(configdir, release, 'sources.list') 

731 if architecture != self.get_system_architecture(): 

732 arch_apt_sources = os.path.join(configdir, release, 

733 architecture, 'sources.list') 

734 if os.path.exists(arch_apt_sources): 

735 apt_sources = arch_apt_sources 

736 

737 # set mirror for get_file_package() 

738 try: 

739 self.set_mirror(self._get_primary_mirror_from_apt_sources(apt_sources)) 

740 except SystemError as e: 

741 apport.warning('cannot determine mirror: %s' % str(e)) 

742 

743 # set current release code name for _distro_release_to_codename 

744 with open(os.path.join(configdir, release, 'codename')) as f: 

745 self.current_release_codename = f.read().strip() 

746 

747 if not os.path.exists(apt_sources): 

748 raise SystemError('%s does not exist' % apt_sources) 

749 

750 # create apt sandbox 

751 if cache_dir: 

752 tmp_aptroot = False 

753 if architecture != self.get_system_architecture(): 

754 aptroot_arch = architecture 

755 else: 

756 aptroot_arch = '' 

757 if configdir: 

758 aptroot = os.path.join(cache_dir, release, aptroot_arch, 'apt') 

759 else: 

760 aptroot = os.path.join(cache_dir, 'system', aptroot_arch, 'apt') 

761 if not os.path.isdir(aptroot):

763 else: 

764 tmp_aptroot = True 

765 aptroot = tempfile.mkdtemp() 

766

768 apt.apt_pkg.config.set('Acquire::Languages', 'none') 

769 # directly connect to Launchpad when downloading deb files 

770 apt.apt_pkg.config.set('Acquire::http::Proxy::api.launchpad.net', 'DIRECT')

823 except KeyError:

879

895 # gdb in artful

898 # apt_pkg.parse_depends needs to handle the or operator,

904 try:

908 # this conflicts with, so we can assume it's not

933 os.unlink(path)

935 del pkg_versions[conflict[0]] 

936 except KeyError:

955 hash="sha1:%s" % sha1sum,

1000 if not pkg_found:

1032 except KeyError:

1057 else:

1059 else:

1062 else:

1071 sys.exit(1) # transient error

1079 # don't extract the same version of the package if it is 

1080 # already extracted

1082 pass 

1083 # don't extract the package if it is a different version than 

1084 # the one we want to extract from Launchpad

1100 for p in pkgs: 

1101 f.write(p) 

1102 f.write(' ') 

1103 f.write(pkg_versions[p]) 

1104 f.write('\n')

1111 + ' '.join(real_pkgs)

1115

1170 def _get_primary_mirror_from_apt_sources(klass, apt_sources):

1208 def _search_contents(self, file, map_cachedir, release, arch): 

1209 '''Internal function for searching file in Contents.gz.''' 

1210 

1211 if map_cachedir: 

1212 dir = map_cachedir 

1213 else: 

1214 if not self._contents_dir: 

1215 self._contents_dir = tempfile.mkdtemp()

1217 

1218 if arch is None: 

1219 arch = self.get_system_architecture() 

1220 if release is None: 

1221 release = self.get_distro_codename() 

1222 else: 

1223 release = self._distro_release_to_codename(release) 

1224 # search -proposed last since file may be provided by a new package 

1225 for pocket in ['-updates', '-security', '', '-proposed']: 

1226 map = os.path.join(dir, '%s%s-Contents-%s.gz' % (release, pocket, arch)) 

1227 

1228 # check if map exists and is younger than a day; if not, we need to 

1229 # refresh it 

1230 try: 

1231 st = os.stat(map) 

1232 age = int(time.time() - st.st_mtime) 

1233 except OSError: 

1234 age = None 

1235 

1236 if age is None or age >= 86400: 

1237 url = '%s/dists/%s%s/Contents-%s.gz' % (self._get_mirror(), release, pocket, arch) 

1238 if age: 

1239 try: 

1240 from httplib import HTTPConnection 

1241 from urlparse import urlparse 

1242 except ImportError: 

1243 # python 3 

1244 from http.client import HTTPConnection 

1245 from urllib.parse import urlparse 

1246 from datetime import datetime 

1247 # HTTPConnection requires server name e.g. 

1248 # archive.ubuntu.com 

1249 server = urlparse(url)[1] 

1250 conn = HTTPConnection(server) 

1251 conn.request("HEAD", urlparse(url)[2]) 

1252 res = conn.getresponse() 

1253 modified_str = res.getheader('last-modified', None) 

1254 if modified_str:

1311 def create_ppa_source_from_origin(klass, origin, distro, release_codename): 

1312 '''For an origin from a Launchpad PPA create sources.list content. 

1313 

1314 distro is the distribution for which content is being created e.g. 

1315 ubuntu. 

1316 

1317 release_codename is the codename of the release for which content is 

1318 being created e.g. trusty. 

1319 

1320 Return a string containing content suitable for writing to a sources.list 

1321 file, or None if the origin is not a Launchpad PPA. 

1322 ''' 

1323 

1324 if origin.startswith("LP-PPA-"): 

1325 components = origin.split("-")[2:] 

1326 # If the PPA is unnamed, it will not appear in origin information 

1327 # but is named ppa in Launchpad. 

1328 try_ppa = True 

1329 if len(components) == 1: 

1330 components.append('ppa') 

1331 try_ppa = False 

1332 

1333 index = 1 

1334 while index < len(components): 

1335 # For an origin we can't tell where the user name ends and the 

1336 # PPA name starts, so split on each "-" until we find a PPA 

1337 # that exists. 

1338 user = str.join('-', components[:index]) 

1339 ppa_name = str.join('-', components[index:]) 

1340 try: 

1341 with closing(urlopen(apport.packaging._ppa_archive_url % 

1342 {'user': user, 'distro': distro, 

1343 'ppaname': ppa_name})) as response: 

1344 response.read() 

1345 # in Python 2 an error does not raise an exception 

1346 if response.getcode() >= 400: 

1347 raise HTTPError('%u' % response.getcode()) 

1348 except (URLError, HTTPError): 

1349 index += 1 

1350 if index == len(components): 

1351 if try_ppa: 

1352 components.append('ppa') 

1353 try_ppa = False 

1354 index = 2 

1355 else: 

1356 user = None 

1357 continue 

1358 break 

1359 if user and ppa_name: 

1360 ppa_line = 'deb http://ppa.launchpad.net/%s/%s/%s %s main' % \ 

1361 (user, ppa_name, distro, release_codename) 

1362 debug_url = 'http://ppa.launchpad.net/%s/%s/%s/dists/%s/main/debug' % \ 

1363 (user, ppa_name, distro, release_codename) 

1364 try: 

1365 with closing(urlopen(debug_url)) as response: 

1366 response.read() 

1367 # in Python 2 an error does not raise an exception 

1368 if response.getcode() >= 400: 

1369 raise HTTPError('%u' % response.getcode()) 

1370 add_debug = ' main/debug' 

1371 except (URLError, HTTPError): 

1372 add_debug = '' 

1373 return ppa_line + add_debug + '\ndeb-src' + ppa_line[3:] + '\n' 

1374 return None
