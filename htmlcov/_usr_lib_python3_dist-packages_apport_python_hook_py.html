19def enabled(): 

20 '''Return whether Apport should generate crash reports.''' 

21 

22 # This doesn't use apport.packaging.enabled() because it is too heavyweight 

23 # See LP: #528355 

24 import re 

25 try: 

26 with open(CONFIG) as f: 

27 conf = f.read() 

28 return re.search(r'^\s*enabled\s*=\s*0\s*$', conf, re.M) is None 

29 except IOError: 

30 # if the file does not exist, assume it's enabled 

31 return True

34def apport_excepthook(exc_type, exc_obj, exc_tb): 

35 '''Catch an uncaught exception and make a traceback.''' 

36 

37 # create and save a problem report. Note that exceptions in this code 

38 # are bad, and we probably need a per-thread reentrancy guard to 

39 # prevent that happening. However, on Ubuntu there should never be 

40 # a reason for an exception here, other than [say] a read only var 

41 # or some such. So what we do is use a try - finally to ensure that 

42 # the original excepthook is invoked, and until we get bug reports 

43 # ignore the other issues. 

44 

45 # import locally here so that there is no routine overhead on python 

46 # startup time - only when a traceback occurs will this trigger. 

47 try: 

48 # ignore 'safe' exit types. 

49 if exc_type in (KeyboardInterrupt, ): 

50 return 

51 

52 # do not do anything if apport was disabled 

53 if not enabled(): 

54 return 

55 

56 try: 

57 from cStringIO import StringIO 

58 StringIO # pyflakes 

59 except ImportError: 

60 from io import StringIO 

61 

62 import re, traceback 

63 from apport.fileutils import likely_packaged, get_recent_crashes 

64 

65 # apport will look up the package from the executable path. 

66 try: 

67 binary = os.path.realpath(os.path.join(os.getcwd(), sys.argv[0])) 

68 except (TypeError, AttributeError, IndexError): 

69 # the module has mutated sys.argv, plan B 

70 try: 

71 binary = os.readlink('/proc/%i/exe' % os.getpid()) 

72 except OSError: 

73 return 

74 

75 # for interactive python sessions, sys.argv[0] == ''; catch that and 

76 # other irregularities 

77 if not os.access(binary, os.X_OK) or not os.path.isfile(binary): 

78 return 

79 

80 # filter out binaries in user accessible paths 

81 if not likely_packaged(binary): 

82 return 

83 

84 import apport.report 

85 

86 pr = apport.report.Report() 

87 

88 # special handling of dbus-python exceptions 

89 if hasattr(exc_obj, 'get_dbus_name'): 

90 name = exc_obj.get_dbus_name() 

91 if name == 'org.freedesktop.DBus.Error.NoReply': 

92 # NoReply is an useless crash, we do not even get the method it 

93 # was trying to call; needs actual crash from D-BUS backend (LP #914220) 

94 return 

95 elif name == 'org.freedesktop.DBus.Error.ServiceUnknown': 

96 dbus_service_unknown_analysis(exc_obj, pr) 

97 else: 

98 pr['_PythonExceptionQualifier'] = name 

99 

100 # disambiguate OSErrors with errno: 

101 if exc_type == OSError and exc_obj.errno is not None: 

102 pr['_PythonExceptionQualifier'] = str(exc_obj.errno) 

103 

104 # append a basic traceback. In future we may want to include 

105 # additional data such as the local variables, loaded modules etc. 

106 tb_file = StringIO() 

107 traceback.print_exception(exc_type, exc_obj, exc_tb, file=tb_file) 

108 pr['Traceback'] = tb_file.getvalue().strip() 

109 pr.add_proc_info(extraenv=['PYTHONPATH', 'PYTHONHOME']) 

110 pr.add_user_info() 

111 # override the ExecutablePath with the script that was actually running 

112 pr['ExecutablePath'] = binary 

113 if 'ExecutableTimestamp' in pr: 

114 pr['ExecutableTimestamp'] = str(int(os.stat(binary).st_mtime)) 

115 try: 

116 pr['PythonArgs'] = '%r' % sys.argv 

117 except AttributeError: 

118 pass 

119 if pr.check_ignored(): 

120 return 

121 mangled_program = re.sub('/', '_', binary) 

122 # get the uid for now, user name later 

123 user = os.getuid() 

124 pr_filename = '%s/%s.%i.crash' % (os.environ.get( 

125 'APPORT_REPORT_DIR', '/var/crash'), mangled_program, user) 

126 crash_counter = 0 

127 if os.path.exists(pr_filename): 

128 if apport.fileutils.seen_report(pr_filename): 

129 # flood protection 

130 with open(pr_filename, 'rb') as f: 

131 crash_counter = get_recent_crashes(f) + 1 

132 if crash_counter > 1: 

133 return 

134 

135 # remove the old file, so that we can create the new one with 

136 # os.O_CREAT|os.O_EXCL 

137 os.unlink(pr_filename) 

138 else: 

139 # don't clobber existing report 

140 return 

141 

142 if crash_counter: 

143 pr['CrashCounter'] = str(crash_counter) 

144 with os.fdopen(os.open(pr_filename, 

145 os.O_WRONLY | os.O_CREAT | os.O_EXCL, 0o640), 'wb') as f: 

146 pr.write(f) 

147 

148 finally: 

149 # resume original processing to get the default behaviour, 

150 # but do not trigger an AttributeError on interpreter shutdown. 

151 if sys: 

152 sys.__excepthook__(exc_type, exc_obj, exc_tb)

155def dbus_service_unknown_analysis(exc_obj, report): 

156 from glob import glob 

157 import subprocess, re 

158 try: 

159 from configparser import ConfigParser, NoSectionError, NoOptionError 

160 (ConfigParser, NoSectionError, NoOptionError) # pyflakes 

161 except ImportError: 

162 # Python 2 

163 from ConfigParser import ConfigParser, NoSectionError, NoOptionError 

164 

165 # determine D-BUS name 

166 m = re.search(r'name\s+(\S+)\s+was not provided by any .service', 

167 exc_obj.get_dbus_message()) 

168 if not m: 

169 if sys.stderr: 

170 sys.stderr.write('Error: cannot parse D-BUS name from exception: ' + 

171 exc_obj.get_dbus_message()) 

172 return 

173 

174 dbus_name = m.group(1) 

175 

176 # determine .service file and Exec name for the D-BUS name 

177 services = [] # tuples of (service file, exe name, running) 

178 for f in glob('/usr/share/dbus-1/*services/*.service'): 

179 cp = ConfigParser(interpolation=None) 

180 cp.read(f, encoding='UTF-8') 

181 try: 

182 if cp.get('D-BUS Service', 'Name') == dbus_name: 

183 exe = cp.get('D-BUS Service', 'Exec') 

184 running = (subprocess.call(['pidof', '-sx', exe], stdout=subprocess.PIPE) == 0) 

185 services.append((f, exe, running)) 

186 except (NoSectionError, NoOptionError): 

187 if sys.stderr: 

188 sys.stderr.write('Invalid D-BUS .service file %s: %s' % ( 

189 f, exc_obj.get_dbus_message())) 

190 continue 

191 

192 if not services: 

193 report['DbusErrorAnalysis'] = 'no service file providing ' + dbus_name 

194 else: 

195 report['DbusErrorAnalysis'] = 'provided by' 

196 for (service, exe, running) in services: 

197 report['DbusErrorAnalysis'] += ' %s (%s is %srunning)' % ( 

198 service, exe, ('' if running else 'not '))
