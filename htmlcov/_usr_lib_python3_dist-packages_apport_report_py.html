115def _check_bug_pattern(report, pattern): 

116 '''Check if given report matches the given bug pattern XML DOM node. 

117 

118 Return the bug URL on match, otherwise None. 

119 ''' 

120 if _python2: 

121 if not pattern.attributes.has_key('url'): 

122 return None 

123 else: 

124 if 'url' not in pattern.attributes: 

125 return None 

126 

127 for c in pattern.childNodes: 

128 # regular expression condition 

129 if c.nodeType == xml.dom.Node.ELEMENT_NODE and c.nodeName == 're': 

130 try: 

131 key = c.attributes['key'].nodeValue 

132 except KeyError: 

133 continue 

134 if key not in report: 

135 return None 

136 c.normalize() 

137 if c.hasChildNodes() and c.childNodes[0].nodeType == xml.dom.Node.TEXT_NODE: 

138 regexp = c.childNodes[0].nodeValue 

139 if _python2: 

140 regexp = regexp.encode('UTF-8') 

141 v = report[key] 

142 if isinstance(v, problem_report.CompressedValue): 

143 v = v.get_value() 

144 if not _python2: 

145 regexp = regexp.encode('UTF-8') 

146 elif isinstance(v, bytes): 

147 if not _python2: 

148 regexp = regexp.encode('UTF-8') 

149 try: 

150 re_c = re.compile(regexp) 

151 except Exception: 

152 continue 

153 if not re_c.search(v): 

154 return None 

155 

156 if _python2: 

157 return pattern.attributes['url'].nodeValue.encode('UTF-8') 

158 else: 

159 return pattern.attributes['url'].nodeValue

190 if not os.path.exists(hook): 

191 return False 

192 

193 symb = {} 

194 try: 

195 with open(hook) as fd: 

196 exec(compile(fd.read(), hook, 'exec'), symb) 

197 try: 

198 symb['add_info'](report, ui) 

199 except TypeError as e:

204 else: 

205 raise 

206 except StopIteration:

208 except Exception as e: 

209 hookname = os.path.splitext(os.path.basename(hook))[0].replace('-', '_')

211 apport.error('hook %s crashed:', hook)

381 def _check_interpreted(self): 

382 '''Check if process is a script. 

383 

384 Use ExecutablePath, ProcStatus and ProcCmdline to determine if 

385 process is an interpreted script. If so, set InterpreterPath 

386 accordingly. 

387 ''' 

388 if 'ExecutablePath' not in self: 

389 return 

390 

391 exebasename = os.path.basename(self['ExecutablePath']) 

392 

393 # check if we consider ExecutablePath an interpreter; we have to do 

394 # this, otherwise 'gedit /tmp/foo.txt' would be detected as interpreted 

395 # script as well 

396 if not any(filter(lambda i: fnmatch.fnmatch(exebasename, i), interpreters)): 

397 return 

398 

399 # first, determine process name 

400 name = None 

401 for line in self['ProcStatus'].splitlines(): 

402 try: 

403 (k, v) = line.split('\t', 1) 

404 except ValueError: 

405 continue 

406 if k == 'Name:': 

407 name = v 

408 break 

409 if not name: 

410 return 

411 

412 cmdargs = self['ProcCmdline'].split('\0') 

413 bindirs = ['/bin/', '/sbin/', '/usr/bin/', '/usr/sbin/'] 

414 

415 # filter out interpreter options 

416 while len(cmdargs) >= 2 and cmdargs[1].startswith('-'): 

417 # check for -m 

418 if name.startswith('python') and cmdargs[1] == '-m' and len(cmdargs) >= 3: 

419 path = self._python_module_path(cmdargs[2]) 

420 if path: 

421 self['InterpreterPath'] = self['ExecutablePath'] 

422 self['ExecutablePath'] = path 

423 else: 

424 self['UnreportableReason'] = 'Cannot determine path of python module %s' % cmdargs[2] 

425 return 

426 

427 del cmdargs[1] 

428 

429 # catch scripts explicitly called with interpreter 

430 if len(cmdargs) >= 2: 

431 # ensure that cmdargs[1] is an absolute path 

432 if cmdargs[1].startswith('.') and 'ProcCwd' in self: 

433 cmdargs[1] = os.path.join(self['ProcCwd'], cmdargs[1]) 

434 if os.access(cmdargs[1], os.R_OK): 

435 self['InterpreterPath'] = self['ExecutablePath'] 

436 self['ExecutablePath'] = os.path.realpath(cmdargs[1]) 

437 

438 # catch directly executed scripts 

439 if 'InterpreterPath' not in self and name != exebasename: 

440 for p in bindirs: 

441 if os.access(p + cmdargs[0], os.R_OK): 

442 argvexe = p + cmdargs[0] 

443 if os.path.basename(os.path.realpath(argvexe)) == name: 

444 self['InterpreterPath'] = self['ExecutablePath'] 

445 self['ExecutablePath'] = argvexe 

446 break 

447 

448 # special case: crashes from twistd are usually the fault of the 

449 # launched program 

450 if 'InterpreterPath' in self and os.path.basename(self['ExecutablePath']) == 'twistd': 

451 self['InterpreterPath'] = self['ExecutablePath'] 

452 exe = self._twistd_executable() 

453 if exe: 

454 self['ExecutablePath'] = exe 

455 else: 

456 self['UnreportableReason'] = 'Cannot determine twistd client program'

458 def _twistd_executable(self): 

459 '''Determine the twistd client program from ProcCmdline.''' 

460 

461 args = self['ProcCmdline'].split('\0')[2:] 

462 

463 # search for a -f/--file, -y/--python or -s/--source argument 

464 while args: 

465 arg = args[0].split('=', 1) 

466 if arg[0].startswith('--file') or arg[0].startswith('--python') or arg[0].startswith('--source'): 

467 if len(arg) == 2: 

468 return arg[1] 

469 else: 

470 return args[1] 

471 elif len(arg[0]) > 1 and arg[0][0] == '-' and arg[0][1] != '-': 

472 opts = arg[0][1:] 

473 if 'f' in opts or 'y' in opts or 's' in opts: 

474 return args[1] 

475 

476 args.pop(0) 

477 

478 return None

482 '''Determine path of given Python module''' 

483 

484 module = module.replace('/', '.').split('.') 

485 pathlist = sys.path 

486 

487 path = None 

488 while module: 

489 name = module.pop(0) 

490

501 module = ['__init__'] 

502 

503 if path and path.endswith('.pyc'): 

504 path = path[:-1] 

505 return path

507 def add_proc_info(self, pid=None, extraenv=[]): 

508 '''Add /proc/pid information. 

509 

510 If neither pid nor self.pid are given, it defaults to the process' 

511 current pid and sets self.pid. 

512 

513 This adds the following fields: 

514 - ExecutablePath: /proc/pid/exe contents; if the crashed process is 

515 interpreted, this contains the script path instead 

516 - InterpreterPath: /proc/pid/exe contents if the crashed process is 

517 interpreted; otherwise this key does not exist 

518 - ExecutableTimestamp: time stamp of ExecutablePath, for comparing at 

519 report time 

520 - ProcEnviron: A subset of the process' environment (only some standard 

521 variables that do not disclose potentially sensitive information, plus 

522 the ones mentioned in extraenv) 

523 - ProcCmdline: /proc/pid/cmdline contents 

524 - ProcStatus: /proc/pid/status contents 

525 - ProcMaps: /proc/pid/maps contents 

526 - ProcAttrCurrent: /proc/pid/attr/current contents, if not "unconfined" 

527 - CurrentDesktop: Value of $XDG_CURRENT_DESKTOP, if present 

528 - _LogindSession: logind cgroup path, if present (Used for filtering 

529 out crashes that happened in a session that is not running any more) 

530 ''' 

531 if not pid: 

532 pid = self.pid or os.getpid() 

533 if not self.pid: 

534 self.pid = int(pid) 

535 pid = str(pid) 

536 

537 try: 

538 self['ProcCwd'] = os.readlink('/proc/' + pid + '/cwd') 

539 except OSError: 

540 pass 

541 self.add_proc_environ(pid, extraenv) 

542 self['ProcStatus'] = _read_file('/proc/' + pid + '/status') 

543 self['ProcCmdline'] = _read_file('/proc/' + pid + '/cmdline').rstrip('\0') 

544 self['ProcMaps'] = _read_maps(int(pid)) 

545 try: 

546 self['ExecutablePath'] = os.readlink('/proc/' + pid + '/exe') 

547 except OSError as e: 

548 if e.errno == errno.ENOENT: 

549 raise ValueError('invalid process') 

550 else: 

551 raise 

552 for p in ('rofs', 'rwfs', 'squashmnt', 'persistmnt'): 

553 if self['ExecutablePath'].startswith('/%s/' % p): 

554 self['ExecutablePath'] = self['ExecutablePath'][len('/%s' % p):] 

555 break 

556 if not os.path.exists(self['ExecutablePath']): 

557 raise ValueError('%s does not exist' % self['ExecutablePath']) 

558 

559 # check if we have an interpreted program 

560 self._check_interpreted() 

561 

562 self['ExecutableTimestamp'] = str(int(os.stat(self['ExecutablePath']).st_mtime)) 

563 

564 # make ProcCmdline ASCII friendly, do shell escaping 

565 self['ProcCmdline'] = self['ProcCmdline'].replace('\\', '\\\\').replace(' ', '\\ ').replace('\0', ' ') 

566 

567 # grab AppArmor or SELinux context 

568 # If no LSM is loaded, reading will return -EINVAL 

569 try: 

570 # On Linux 2.6.28+, 'current' is world readable, but read() gives 

571 # EPERM; Python 2.5.3+ crashes on that (LP: #314065) 

572 if os.getuid() == 0: 

573 with open('/proc/' + pid + '/attr/current') as fd: 

574 val = fd.read().strip() 

575 if val != 'unconfined': 

576 self['ProcAttrCurrent'] = val 

577 except (IOError, OSError): 

578 pass 

579 

580 ret = self.get_logind_session(pid) 

581 if ret: 

582 self['_LogindSession'] = ret[0]

585 '''Add environment information. 

586 

587 If pid is not given, it defaults to the process' current pid. 

588 

589 This adds the following fields:

622 if self['ProcEnviron']: 

623 self['ProcEnviron'] += '\n' 

624 self['ProcEnviron'] += 'PATH=(custom, no user)' 

625 elif line.startswith('XDG_RUNTIME_DIR='): 

626 if self['ProcEnviron']: 

627 self['ProcEnviron'] += '\n' 

628 self['ProcEnviron'] += 'XDG_RUNTIME_DIR=<set>' 

629 elif line.startswith('LD_PRELOAD='): 

630 if self['ProcEnviron']:

664 stderr=subprocess.STDOUT) 

665 except OSError:

667 p.stdin.write('bt -a -f\n') 

668 p.stdin.write('ps\n') 

669 p.stdin.write('runq\n') 

670 p.stdin.write('quit\n')

674 if ret:

676 finally: 

677 if unlink_core: 

678 os.unlink(core) 

679 return ret

681 def add_gdb_info(self, rootdir=None, gdb_sandbox=None): 

682 '''Add information from gdb. 

683 

684 This requires that the report has a CoreDump and an 

685 ExecutablePath. This adds the following fields: 

686 - Registers: Output of gdb's 'info registers' command 

687 - Disassembly: Output of gdb's 'x/16i $pc' command 

688 - Stacktrace: Output of gdb's 'bt full' command 

689 - ThreadStacktrace: Output of gdb's 'thread apply all bt full' command 

690 - StacktraceTop: simplified stacktrace (topmost 5 functions) for inline 

691 inclusion into bug reports and easier processing 

692 - AssertionMessage: Value of __abort_msg, __glib_assert_msg, or 

693 __nih_abort_msg if present 

694 

695 The optional rootdir can specify a root directory which has the 

696 executable, libraries, and debug symbols. This does not require 

697 chroot() or root privileges, it just instructs gdb to search for the 

698 files there. 

699 

700 Raises a IOError if the core dump is invalid/truncated, or OSError if 

701 calling gdb fails. 

702 ''' 

703 if 'CoreDump' not in self or 'ExecutablePath' not in self: 

704 return 

705 

706 gdb_reports = {'Registers': 'info registers', 

707 'Disassembly': 'x/16i $pc', 

708 'Stacktrace': 'bt full', 

709 'ThreadStacktrace': 'thread apply all bt full', 

710 'AssertionMessage': 'print __abort_msg->msg', 

711 'GLibAssertionMessage': 'print __glib_assert_msg', 

712 'NihAssertionMessage': 'print (char*) __nih_abort_msg'} 

713 gdb_cmd, environ = self.gdb_command(rootdir, gdb_sandbox) 

714 

715 # limit maximum backtrace depth (to avoid looped stacks) 

716 gdb_cmd += ['--batch', '--ex', 'set backtrace limit 2000'] 

717 

718 value_keys = [] 

719 # append the actual commands and something that acts as a separator 

720 for name, cmd in gdb_reports.items(): 

721 value_keys.append(name) 

722 gdb_cmd += ['--ex', 'p -99', '--ex', cmd] 

723 

724 out = _command_output(gdb_cmd, env=environ).decode('UTF-8', errors='replace') 

725 

726 # check for truncated stack trace 

727 if 'is truncated: expected core file size' in out: 

728 if 'warning:' in out: 

729 warnings = '\n'.join([l for l in out.splitlines() if 'warning:' in l]) 

730 elif 'Warning:' in out: 

731 warnings = '\n'.join([l for l in out.splitlines() if 'Warning:' in l]) 

732 reason = 'Invalid core dump: ' + warnings.strip() 

733 self['UnreportableReason'] = reason 

734 raise IOError(reason) 

735 

736 # split the output into the various fields 

737 part_re = re.compile(r'^\$\d+\s*=\s*-99$', re.MULTILINE) 

738 parts = part_re.split(out) 

739 # drop the gdb startup text prior to first separator 

740 parts.pop(0) 

741 for part in parts: 

742 self[value_keys.pop(0)] = part.replace('\n\n', '\n.\n').strip() 

743 

744 # glib's assertion has precedence, since it internally uses 

745 # abort(), and then glib's __abort_msg is bogus 

746 if 'GLibAssertionMessage' in self: 

747 if '"ERROR:' in self['GLibAssertionMessage']: 

748 self['AssertionMessage'] = self['GLibAssertionMessage'] 

749 del self['GLibAssertionMessage'] 

750 

751 # same reason for libnih's assertion messages 

752 if 'NihAssertionMessage' in self: 

753 if self['NihAssertionMessage'].startswith('$'): 

754 self['AssertionMessage'] = self['NihAssertionMessage'] 

755 del self['NihAssertionMessage'] 

756 

757 # clean up AssertionMessage 

758 if 'AssertionMessage' in self: 

759 # chop off "$n = 0x...." prefix, drop empty ones 

760 m = re.match(r'^\$\d+\s+=\s+0x[0-9a-fA-F]+\s+"(.*)"\s*$', 

761 self['AssertionMessage']) 

762 if m: 

763 self['AssertionMessage'] = m.group(1) 

764 if self['AssertionMessage'].endswith('\\n'): 

765 self['AssertionMessage'] = self['AssertionMessage'][0:-2] 

766 else: 

767 del self['AssertionMessage'] 

768 

769 if 'Stacktrace' in self: 

770 self._gen_stacktrace_top() 

771 addr_signature = self.crash_signature_addresses() 

772 if addr_signature: 

773 self['StacktraceAddressSignature'] = addr_signature

775 def _gen_stacktrace_top(self): 

776 '''Build field StacktraceTop as the top five functions of Stacktrace. 

777 

778 Signal handler invocations and related functions are skipped since they 

779 are generally not useful for triaging and duplicate detection. 

780 ''' 

781 unwind_functions = set(['g_logv', 'g_log', 'IA__g_log', 'IA__g_logv', 

782 'g_assert_warning', 'IA__g_assert_warning', 

783 '__GI_abort', '_XError']) 

784 toptrace = [''] * 5 

785 depth = 0 

786 unwound = False 

787 unwinding = False 

788 unwinding_xerror = False 

789 bt_fn_re = re.compile(r'^#(\d+)\s+(?:0x(?:\w+)\s+in\s+\*?(.*)|(<signal handler called>)\s*)$') 

790 bt_fn_noaddr_re = re.compile(r'^#(\d+)\s+(?:(.*)|(<signal handler called>)\s*)$') 

791 # some internal functions like the SSE stubs cause unnecessary jitter 

792 ignore_functions_re = re.compile(r'^(__.*_s?sse\d+(?:_\w+)?|__kernel_vsyscall)$') 

793 

794 for line in self['Stacktrace'].splitlines(): 

795 m = bt_fn_re.match(line) 

796 if not m: 

797 m = bt_fn_noaddr_re.match(line) 

798 if not m: 

799 continue 

800 

801 if not unwound or unwinding: 

802 if m.group(2): 

803 fn = m.group(2).split()[0].split('(')[0] 

804 else: 

805 fn = None 

806 

807 # handle XErrors 

808 if unwinding_xerror: 

809 if fn.startswith('_X') or fn in ['handle_response', 'handle_error', 'XWindowEvent']: 

810 continue 

811 else: 

812 unwinding_xerror = False 

813 

814 if m.group(3) or fn in unwind_functions: 

815 unwinding = True 

816 depth = 0 

817 toptrace = [''] * 5 

818 if m.group(3): 

819 # we stop unwinding when we found a <signal handler>, 

820 # but we continue unwinding otherwise, as e. g. a glib 

821 # abort is usually sitting on top of an XError 

822 unwound = True 

823 

824 if fn == '_XError': 

825 unwinding_xerror = True 

826 continue 

827 else: 

828 unwinding = False 

829 

830 frame = m.group(2) or m.group(3) 

831 function = frame.split()[0] 

832 if depth < len(toptrace) and not ignore_functions_re.match(function): 

833 toptrace[depth] = frame 

834 depth += 1 

835 self['StacktraceTop'] = '\n'.join(toptrace).strip()

978 def check_ignored(self): 

979 '''Check if current report should not be presented. 

980 

981 Reports can be suppressed by per-user blacklisting in 

982 ~/.apport-ignore.xml (in the real UID's home) and 

983 /etc/apport/blacklist.d/. For environments where you are only 

984 interested in crashes of some programs, you can also create a whitelist 

985 in /etc/apport/whitelist.d/, everything which does not match gets 

986 ignored then. 

987 

988 This requires the ExecutablePath attribute. Throws a ValueError if the 

989 file has an invalid format. 

990 ''' 

991 assert 'ExecutablePath' in self 

992 

993 # check blacklist 

994 try: 

995 for f in os.listdir(_blacklist_dir): 

996 try: 

997 with open(os.path.join(_blacklist_dir, f)) as fd: 

998 for line in fd: 

999 if line.strip() == self['ExecutablePath']: 

1000 return True 

1001 except IOError: 

1002 continue 

1003 except OSError: 

1004 pass 

1005 

1006 # check whitelist 

1007 try: 

1008 whitelist = set() 

1009 for f in os.listdir(_whitelist_dir): 

1010 try: 

1011 with open(os.path.join(_whitelist_dir, f)) as fd: 

1012 for line in fd: 

1013 whitelist.add(line.strip()) 

1014 except IOError: 

1015 continue 

1016 

1017 if whitelist and self['ExecutablePath'] not in whitelist: 

1018 return True 

1019 except OSError: 

1020 pass 

1021 

1022 try: 

1023 dom = self._get_ignore_dom() 

1024 except (ValueError, KeyError): 

1025 apport.error('Could not get ignore file:') 

1026 traceback.print_exc() 

1027 return False 

1028 

1029 try: 

1030 cur_mtime = int(os.stat(self['ExecutablePath']).st_mtime) 

1031 except OSError: 

1032 # if it does not exist any more, do nothing 

1033 return False 

1034 

1035 # search for existing entry and update it 

1036 for ignore in dom.getElementsByTagName('ignore'): 

1037 if ignore.getAttribute('program') == self['ExecutablePath']: 

1038 if float(ignore.getAttribute('mtime')) >= cur_mtime: 

1039 return True 

1040 

1041 return False

1044 '''Ignore future crashes of this executable. 

1045 

1046 Add a ignore list entry for this report to ~/.apport-ignore.xml, so 

1047 that future reports for this ExecutablePath are not presented to the 

1048 user any more. 

1049 

1050 Throws a ValueError if the file already exists and has an invalid 

1051 format. 

1052 ''' 

1053 assert 'ExecutablePath' in self 

1054

1060 if e.errno == errno.ENOENT: 

1061 return

1118 def standard_title(self): 

1119 '''Create an appropriate title for a crash database entry. 

1120 

1121 This contains the topmost function name from the stack trace and the 

1122 signal (for signal crashes) or the Python exception (for unhandled 

1123 Python exceptions). 

1124 

1125 Return None if the report is not a crash or a default title could not 

1126 be generated. 

1127 ''' 

1128 # assertion failure 

1129 if self.get('Signal') == '6' and \ 

1130 'ExecutablePath' in self and \ 

1131 'AssertionMessage' in self: 

1132 return '%s assert failure: %s' % ( 

1133 os.path.basename(self['ExecutablePath']), 

1134 self['AssertionMessage']) 

1135 

1136 # signal crash 

1137 if 'Signal' in self and 'ExecutablePath' in self and 'StacktraceTop' in self: 

1138 

1139 signal_names = { 

1140 '4': 'SIGILL', 

1141 '6': 'SIGABRT', 

1142 '8': 'SIGFPE', 

1143 '11': 'SIGSEGV', 

1144 '13': 'SIGPIPE'} 

1145 

1146 fn = self.stacktrace_top_function() 

1147 if fn: 

1148 fn = ' in %s()' % fn 

1149 else: 

1150 fn = '' 

1151 

1152 arch_mismatch = '' 

1153 if 'Architecture' in self and 'PackageArchitecture' in self and self['Architecture'] != self['PackageArchitecture'] and self['PackageArchitecture'] != 'all': 

1154 arch_mismatch = ' [non-native %s package]' % self['PackageArchitecture'] 

1155 

1156 return '%s crashed with %s%s%s' % ( 

1157 os.path.basename(self['ExecutablePath']), 

1158 signal_names.get(self.get('Signal'), 'signal ' + self.get('Signal')), 

1159 fn, arch_mismatch 

1160 ) 

1161 

1162 # Python exception 

1163 if 'Traceback' in self and 'ExecutablePath' in self: 

1164 

1165 trace = self['Traceback'].splitlines() 

1166 

1167 if len(trace) < 1: 

1168 return None 

1169 if len(trace) < 3: 

1170 return '%s crashed with %s' % ( 

1171 os.path.basename(self['ExecutablePath']), 

1172 trace[0]) 

1173 

1174 trace_re = re.compile(r'^\s*File\s*"(\S+)".* in (.+)$') 

1175 i = len(trace) - 1 

1176 function = 'unknown' 

1177 while i >= 0: 

1178 m = trace_re.match(trace[i]) 

1179 if m: 

1180 module_path = m.group(1) 

1181 function = m.group(2) 

1182 break 

1183 i -= 1 

1184 

1185 path = os.path.basename(self['ExecutablePath']) 

1186 last_line = trace[-1] 

1187 exception = last_line.split(':')[0] 

1188 m = re.match('^%s: (.+)$' % re.escape(exception), last_line) 

1189 if m: 

1190 message = m.group(1) 

1191 else: 

1192 message = None 

1193 

1194 if function == '<module>': 

1195 if module_path == self['ExecutablePath']: 

1196 context = '__main__' 

1197 else: 

1198 # Maybe use os.path.basename? 

1199 context = module_path 

1200 else: 

1201 context = '%s()' % function 

1202 

1203 title = '%s crashed with %s in %s' % ( 

1204 path, 

1205 exception, 

1206 context 

1207 ) 

1208 

1209 if message: 

1210 title += ': %s' % message 

1211 

1212 return title 

1213 

1214 # package problem 

1215 if self.get('ProblemType') == 'Package' and 'Package' in self: 

1216 

1217 title = 'package %s failed to install/upgrade' % \ 

1218 self['Package'] 

1219 if self.get('ErrorMessage'): 

1220 title += ': ' + self['ErrorMessage'].splitlines()[-1] 

1221 

1222 return title 

1223 

1224 if self.get('ProblemType') == 'KernelOops' and 'OopsText' in self: 

1225 

1226 oops = self['OopsText'] 

1227 if oops.startswith('------------[ cut here ]------------'): 

1228 title = oops.split('\n', 2)[1] 

1229 else: 

1230 title = oops.split('\n', 1)[0] 

1231 

1232 return title 

1233 

1234 if self.get('ProblemType') == 'KernelOops' and 'Failure' in self: 

1235 # Title the report with suspend or hibernate as appropriate, 

1236 # and mention any non-free modules loaded up front. 

1237 title = '' 

1238 if 'MachineType' in self: 

1239 title += '[' + self['MachineType'] + '] ' 

1240 title += self['Failure'] + ' failure' 

1241 if 'NonfreeKernelModules' in self: 

1242 title += ' [non-free: ' + self['NonfreeKernelModules'] + ']' 

1243 title += '\n' 

1244 

1245 return title 

1246 

1247 return None

1262 def crash_signature(self): 

1263 '''Get a signature string for a crash. 

1264 

1265 This is suitable for identifying duplicates. 

1266 

1267 For signal crashes this the concatenation of ExecutablePath, Signal 

1268 number, and StacktraceTop function names, separated by a colon. If 

1269 StacktraceTop has unknown functions or the report lacks any of those 

1270 fields, return None. In this case, you can use 

1271 crash_signature_addresses() to get a less precise duplicate signature 

1272 based on addresses instead of symbol names. 

1273 

1274 For assertion failures, it is the concatenation of ExecutablePath 

1275 and assertion message, separated by colons. 

1276 

1277 For Python crashes, this concatenates the ExecutablePath, exception 

1278 name, and Traceback function names, again separated by a colon. 

1279 

1280 For suspend/resume failures, this concatenates whether it was a suspend 

1281 or resume failure with the hardware identifier and the BIOS version, if 

1282 it exists. 

1283 ''' 

1284 if 'ExecutablePath' not in self: 

1285 if not self['ProblemType'] in ('KernelCrash', 'KernelOops'): 

1286 return None 

1287 

1288 # kernel crash 

1289 if 'Stacktrace' in self and self['ProblemType'] == 'KernelCrash': 

1290 sig = 'kernel' 

1291 regex = re.compile(r'^\s*\#\d+\s\[\w+\]\s(\w+)') 

1292 for line in self['Stacktrace'].splitlines(): 

1293 m = regex.match(line) 

1294 if m: 

1295 sig += ':' + (m.group(1)) 

1296 return sig 

1297 

1298 # assertion failures 

1299 if self.get('Signal') == '6' and 'AssertionMessage' in self: 

1300 sig = self['ExecutablePath'] + ':' + self['AssertionMessage'] 

1301 # filter out addresses, to help match duplicates more sanely 

1302 return re.sub(r'0x[0-9a-f]{6,}', 'ADDR', sig) 

1303 

1304 # signal crashes 

1305 if 'StacktraceTop' in self and 'Signal' in self: 

1306 sig = '%s:%s' % (self['ExecutablePath'], self['Signal']) 

1307 bt_fn_re = re.compile(r'^(?:([\w:~]+).*|(<signal handler called>)\s*)$') 

1308 

1309 lines = self['StacktraceTop'].splitlines() 

1310 if len(lines) < 2: 

1311 return None 

1312 

1313 for line in lines: 

1314 m = bt_fn_re.match(line) 

1315 if m: 

1316 sig += ':' + (m.group(1) or m.group(2)) 

1317 else: 

1318 # this will also catch ?? 

1319 return None 

1320 return sig 

1321 

1322 # Python crashes 

1323 if 'Traceback' in self: 

1324 trace = self['Traceback'].splitlines() 

1325 

1326 sig = '' 

1327 if len(trace) == 1: 

1328 # sometimes, Python exceptions do not have file references 

1329 m = re.match(r'(\w+): ', trace[0]) 

1330 if m: 

1331 return self['ExecutablePath'] + ':' + m.group(1) 

1332 else: 

1333 return None 

1334 elif len(trace) < 3: 

1335 return None

1337 loc_re = re.compile(r'^\s+File "([^"]+).*line (\d+).*\sin (.*)$') 

1338 for line in trace: 

1339 m = loc_re.match(line) 

1340 if m: 

1341 # if we have a function name, use this; for a a crash 

1342 # outside of a function/method, fall back to the source 

1343 # file location 

1344 if m.group(3) != '<module>': 

1345 sig += ':' + m.group(3) 

1346 else: 

1347 # resolve symlinks for more stable signatures 

1348 f = m.group(1) 

1349 if os.path.islink(f): 

1350 f = os.path.realpath(f) 

1351 sig += ':%s@%s' % (f, m.group(2)) 

1352 

1353 exc_name = trace[-1].split(':')[0] 

1354 try: 

1355 exc_name += '(%s)' % self['_PythonExceptionQualifier'] 

1356 except KeyError: 

1357 pass 

1358 return self['ExecutablePath'] + ':' + exc_name + sig 

1359 

1360 if self['ProblemType'] == 'KernelOops' and 'Failure' in self: 

1361 if 'suspend' in self['Failure'] or 'resume' in self['Failure']: 

1362 # Suspend / resume failure 

1363 sig = self['Failure'] 

1364 if self.get('MachineType'): 

1365 sig += ':%s' % self['MachineType'] 

1366 if self.get('dmi.bios.version'): 

1367 sig += ':%s' % self['dmi.bios.version'] 

1368 return sig

1370 # KernelOops crashes 

1371 if 'OopsText' in self: 

1372 in_trace_body = False 

1373 parts = [] 

1374 for line in self['OopsText'].split('\n'): 

1375 if line.startswith('BUG: unable to handle'): 

1376 parsed = re.search('^BUG: unable to handle (.*) at ', line) 

1377 if parsed: 

1378 match = parsed.group(1) 

1379 assert match, 'could not parse expected problem type line: %s' % line 

1380 parts.append(match) 

1381 

1382 if line.startswith('IP: '): 

1383 match = self._extract_function_and_address(line) 

1384 if match: 

1385 parts.append(match) 

1386 

1387 elif line.startswith('Call Trace:'): 

1388 in_trace_body = True 

1389 

1390 elif in_trace_body: 

1391 match = None 

1392 if line and line[0] == ' ': 

1393 match = self._extract_function_and_address(line) 

1394 if match: 

1395 parts.append(match) 

1396 else: 

1397 in_trace_body = False 

1398 if parts: 

1399 return ':'.join(parts) 

1400 return None

1411 def crash_signature_addresses(self): 

1412 '''Compute heuristic duplicate signature for a signal crash. 

1413 

1414 This should be used if crash_signature() fails, i. e. Stacktrace does 

1415 not have enough symbols. 

1416 

1417 This approach only uses addresses in the stack trace and does not rely 

1418 on symbol resolution. As we can't unwind these stack traces, we cannot 

1419 limit them to the top five frames and thus will end up with several or 

1420 many different signatures for a particular crash. But these can be 

1421 computed and synchronously checked with a crash database at the client 

1422 side, which avoids having to upload and process the full report. So on 

1423 the server-side crash database we will only have to deal with all the 

1424 equivalence classes (i. e. same crash producing a number of possible 

1425 signatures) instead of every single report. 

1426 

1427 Return None when signature cannot be determined. 

1428 ''' 

1429 if 'ProcMaps' not in self or 'Stacktrace' not in self or 'Signal' not in self: 

1430 return None 

1431 

1432 stack = [] 

1433 failed = 0 

1434 for line in self['Stacktrace'].splitlines(): 

1435 if line.startswith('#'): 

1436 addr = line.split()[1] 

1437 if not addr.startswith('0x'): 

1438 continue 

1439 addr = int(addr, 16) # we do want to know about ValueErrors here, so don't catch 

1440 # ignore impossibly low addresses; these are usually artifacts 

1441 # from gdb when not having debug symbols 

1442 if addr < 0x1000: 

1443 continue 

1444 offset = self._address_to_offset(addr) 

1445 if offset: 

1446 # avoid ':' in ELF paths, we use that as separator 

1447 stack.append(offset.replace(':', '..')) 

1448 else: 

1449 failed += 1 

1450 

1451 # stack unwinding chops off ~ 5 functions, and we need some more 

1452 # accuracy because we do not have symbols; but beyond a depth of 15 

1453 # we get too much noise, so we can abort there 

1454 if len(stack) >= 15: 

1455 break 

1456 

1457 # we only accept a small minority (< 20%) of failed resolutions, otherwise we 

1458 # discard 

1459 if failed > 0 and len(stack) / failed < 4: 

1460 return None 

1461 

1462 # we also discard if the trace is too short 

1463 if (failed == 0 and len(stack) < 3) or (failed > 0 and len(stack) < 6): 

1464 return None 

1465 

1466 return '%s:%s:%s' % ( 

1467 self['ExecutablePath'], 

1468 self['Signal'], 

1469 ':'.join(stack))

1471 def anonymize(self): 

1472 '''Remove user identifying strings from the report. 

1473 

1474 This particularly removes the user name, host name, and IPs 

1475 from attributes which contain data read from the environment, and 

1476 removes the ProcCwd attribute completely. 

1477 ''' 

1478 replacements = [] 

1479 if (os.getuid() > 0): 

1480 # do not replace "root" 

1481 p = pwd.getpwuid(os.getuid()) 

1482 if len(p[0]) >= 2: 

1483 replacements.append((re.compile(r'\b%s\b' % re.escape(p[0])), 'username')) 

1484 replacements.append((re.compile(r'\b%s\b' % re.escape(p[5])), '/home/username')) 

1485 

1486 for s in p[4].split(','): 

1487 s = s.strip() 

1488 if len(s) > 2: 

1489 replacements.append((re.compile(r'(\b|\s)%s\b' % re.escape(s)), r'\1User Name')) 

1490 

1491 hostname = os.uname()[1] 

1492 if len(hostname) >= 2: 

1493 replacements.append((re.compile(r'\b%s\b' % re.escape(hostname)), 'hostname'))

1495 try: 

1496 del self['ProcCwd'] 

1497 except KeyError: 

1498 pass

1500 for k in self:

1504 if not hasattr(self[k], 'isspace'): 

1505 continue 

1506 for (pattern, repl) in replacements: 

1507 if type(self[k]) == bytes: 

1508 self[k] = pattern.sub(repl, self[k].decode('UTF-8', errors='replace')).encode('UTF-8')

1603 def _address_to_offset(self, addr): 

1604 '''Resolve a memory address to an ELF name and offset. 

1605 

1606 This can be used for building duplicate signatures from non-symbolic 

1607 stack traces. These often do not have enough symbols available to 

1608 resolve function names, but taking the raw addresses also is not 

1609 suitable due to ASLR. But the offsets within a library should be 

1610 constant between crashes (assuming the same version of all libraries). 

1611 

1612 This needs and uses the "ProcMaps" field to resolve addresses. 

1613 

1614 Return 'path+offset' when found, or None if address is not in any 

1615 mapped range. 

1616 ''' 

1617 self._build_proc_maps_cache() 

1618 

1619 for (start, end, elf) in self._proc_maps_cache: 

1620 if start <= addr and end >= addr: 

1621 return '%s+%x' % (elf, addr - start) 

1622 

1623 return None

1625 def _build_proc_maps_cache(self): 

1626 '''Generate self._proc_maps_cache from ProcMaps field. 

1627 

1628 This only gets done once. 

1629 ''' 

1630 if self._proc_maps_cache: 

1631 return 

1632 

1633 assert 'ProcMaps' in self 

1634 self._proc_maps_cache = [] 

1635 # library paths might have spaces, so we need to make some assumptions 

1636 # about the intermediate fields. But we know that in between the pre-last 

1637 # data field and the path there are many spaces, while between the 

1638 # other data fields there is only one. So we take 2 or more spaces as 

1639 # the separator of the last data field and the path. 

1640 fmt = re.compile(r'^([0-9a-fA-F]+)-([0-9a-fA-F]+).*\s{2,}(\S.*$)') 

1641 fmt_unknown = re.compile(r'^([0-9a-fA-F]+)-([0-9a-fA-F]+)\s') 

1642 

1643 for line in self['ProcMaps'].splitlines(): 

1644 if not line.strip(): 

1645 continue 

1646 m = fmt.match(line) 

1647 if not m: 

1648 # ignore lines with unknown ELF 

1649 if fmt_unknown.match(line): 

1650 continue 

1651 # but complain otherwise, as this means we encounter an 

1652 # architecture or new kernel version where the format changed 

1653 assert m, 'cannot parse ProcMaps line: ' + line 

1654 self._proc_maps_cache.append((int(m.group(1), 16), 

1655 int(m.group(2), 16), m.group(3)))
